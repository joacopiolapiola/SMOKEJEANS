CREATE DATABASE smoke;
USE smoke;
CREATE TABLE categorias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100)
);
CREATE TABLE admins (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    password VARCHAR(255)
);
CREATE TABLE product_images (
    id INT AUTO_INCREMENT PRIMARY KEY,
    file_path VARCHAR(255) NOT NULL
);
CREATE TABLE productos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150),
    descripcion TEXT,
    precio DECIMAL(10,2),
    imagen1 INT NOT NULL,
    imagen2 INT,
    imagen3 INT,
    imagen4 INT,
    categoria_id INT,
    dispo TINYINT(1) DEFAULT 1,
    FOREIGN KEY (imagen1)
        REFERENCES product_images(id),
    FOREIGN KEY (imagen2)
        REFERENCES product_images(id),
    FOREIGN KEY (imagen3)
        REFERENCES product_images(id),
    FOREIGN KEY (imagen4)
        REFERENCES product_images(id)
);
CREATE TABLE prod_categoria (
    prod_id INT NOT NULL,
    categoria_id INT NOT NULL,
    PRIMARY KEY (prod_id, categoria_id),
    FOREIGN KEY (prod_id) REFERENCES productos(id) ON DELETE CASCADE,
    FOREIGN KEY (categoria_id) REFERENCES categorias(id) ON DELETE CASCADE
);
